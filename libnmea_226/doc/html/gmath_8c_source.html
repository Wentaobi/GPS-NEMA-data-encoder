105 double nmea_distance_ellipsoid(
106  const nmeaPOS *from_pos,
107  const nmeaPOS *to_pos,
108  double *from_azimuth,
109  double *to_azimuth
110  )
111 {
112  /* All variables */
113  double f, a, b, sqr_a, sqr_b;
114  double L, phi1, phi2, U1, U2, sin_U1, sin_U2, cos_U1, cos_U2;
115  double sigma, sin_sigma, cos_sigma, cos_2_sigmam, sqr_cos_2_sigmam, sqr_cos_alpha, lambda, sin_lambda, cos_lambda, delta_lambda;
116  int remaining_steps;
117  double sqr_u, A, B, delta_sigma;
118 
119  /* Check input */
120  NMEA_ASSERT(from_pos != 0);
121  NMEA_ASSERT(to_pos != 0);
122 
123  if ((from_pos->lat == to_pos->lat) && (from_pos->lon == to_pos->lon))
124  { /* Identical points */
125  if ( from_azimuth != 0 )
126  *from_azimuth = 0;
127  if ( to_azimuth != 0 )
128  *to_azimuth = 0;
129  return 0;
130  } /* Identical points */
131 
132  /* Earth geometry */
133  f = NMEA_EARTH_FLATTENING;
134  a = NMEA_EARTH_SEMIMAJORAXIS_M;
135  b = (1 - f) * a;
136  sqr_a = a * a;
137  sqr_b = b * b;
138 
139  /* Calculation */
140  L = to_pos->lon - from_pos->lon;
141  phi1 = from_pos->lat;
142  phi2 = to_pos->lat;
143  U1 = atan((1 - f) * tan(phi1));
144  U2 = atan((1 - f) * tan(phi2));
145  sin_U1 = sin(U1);
146  sin_U2 = sin(U2);
147  cos_U1 = cos(U1);
148  cos_U2 = cos(U2);
149 
150  /* Initialize iteration */
151  sigma = 0;
152  sin_sigma = sin(sigma);
153  cos_sigma = cos(sigma);
154  cos_2_sigmam = 0;
155  sqr_cos_2_sigmam = cos_2_sigmam * cos_2_sigmam;
156  sqr_cos_alpha = 0;
157  lambda = L;
158  sin_lambda = sin(lambda);
159  cos_lambda = cos(lambda);
160  delta_lambda = lambda;
161  remaining_steps = 20;
162 
163  while ((delta_lambda > 1e-12) && (remaining_steps > 0))
164  { /* Iterate */
165  /* Variables */
166  double tmp1, tmp2, tan_sigma, sin_alpha, cos_alpha, C, lambda_prev;
167 
168  /* Calculation */
169  tmp1 = cos_U2 * sin_lambda;
170  tmp2 = cos_U1 * sin_U2 - sin_U1 * cos_U2 * cos_lambda;
171  sin_sigma = sqrt(tmp1 * tmp1 + tmp2 * tmp2);
172  cos_sigma = sin_U1 * sin_U2 + cos_U1 * cos_U2 * cos_lambda;
173  tan_sigma = sin_sigma / cos_sigma;
174  sin_alpha = cos_U1 * cos_U2 * sin_lambda / sin_sigma;
175  cos_alpha = cos(asin(sin_alpha));
176  sqr_cos_alpha = cos_alpha * cos_alpha;
177  cos_2_sigmam = cos_sigma - 2 * sin_U1 * sin_U2 / sqr_cos_alpha;
178  sqr_cos_2_sigmam = cos_2_sigmam * cos_2_sigmam;
179  C = f / 16 * sqr_cos_alpha * (4 + f * (4 - 3 * sqr_cos_alpha));
180  lambda_prev = lambda;
181  sigma = asin(sin_sigma);
182  lambda = L +
183  (1 - C) * f * sin_alpha
184  * (sigma + C * sin_sigma * (cos_2_sigmam + C * cos_sigma * (-1 + 2 * sqr_cos_2_sigmam)));
185  delta_lambda = lambda_prev - lambda;
186  if ( delta_lambda < 0 ) delta_lambda = -delta_lambda;
187  sin_lambda = sin(lambda);
188  cos_lambda = cos(lambda);
189  remaining_steps--;
190  } /* Iterate */
191 
192  /* More calculation */
193  sqr_u = sqr_cos_alpha * (sqr_a - sqr_b) / sqr_b;
194  A = 1 + sqr_u / 16384 * (4096 + sqr_u * (-768 + sqr_u * (320 - 175 * sqr_u)));
195  B = sqr_u / 1024 * (256 + sqr_u * (-128 + sqr_u * (74 - 47 * sqr_u)));
196  delta_sigma = B * sin_sigma * (
197  cos_2_sigmam + B / 4 * (
198  cos_sigma * (-1 + 2 * sqr_cos_2_sigmam) -
199  B / 6 * cos_2_sigmam * (-3 + 4 * sin_sigma * sin_sigma) * (-3 + 4 * sqr_cos_2_sigmam)
200  ));
201 
202  /* Calculate result */
203  if ( from_azimuth != 0 )
204  {
205  double tan_alpha_1 = cos_U2 * sin_lambda / (cos_U1 * sin_U2 - sin_U1 * cos_U2 * cos_lambda);
206  *from_azimuth = atan(tan_alpha_1);
207  }
208  if ( to_azimuth != 0 )
209  {
210  double tan_alpha_2 = cos_U1 * sin_lambda / (-sin_U1 * cos_U2 + cos_U1 * sin_U2 * cos_lambda);
211  *to_azimuth = atan(tan_alpha_2);
212  }
213 
214  return b * A * (sigma - delta_sigma);
215 }

220 int nmea_move_horz(
221  const nmeaPOS *start_pos,
222  nmeaPOS *end_pos,
223  double azimuth,
224  double distance
225  )
226 {
227  nmeaPOS p1 = *start_pos;
228  int RetVal = 1;
229 
230  distance /= NMEA_EARTHRADIUS_KM; /* Angular distance covered on earth's surface */
231  azimuth = nmea_degree2radian(azimuth);
232 
233  end_pos->lat = asin(
234  sin(p1.lat) * cos(distance) + cos(p1.lat) * sin(distance) * cos(azimuth));
235  end_pos->lon = p1.lon + atan2(
236  sin(azimuth) * sin(distance) * cos(p1.lat), cos(distance) - sin(p1.lat) * sin(end_pos->lat));
237 
238  if(NMEA_POSIX(isnan)(end_pos->lat) || NMEA_POSIX(isnan)(end_pos->lon))
239  {
240  end_pos->lat = 0; end_pos->lon = 0;
241  RetVal = 0;
242  }
243 
244  return RetVal;
245 }

253 int nmea_move_horz_ellipsoid(
254  const nmeaPOS *start_pos,
255  nmeaPOS *end_pos,
256  double azimuth,
257  double distance,
258  double *end_azimuth
259  )
260 {
261  /* Variables */
262  double f, a, b, sqr_a, sqr_b;
263  double phi1, tan_U1, sin_U1, cos_U1, s, alpha1, sin_alpha1, cos_alpha1;
264  double tan_sigma1, sigma1, sin_alpha, cos_alpha, sqr_cos_alpha, sqr_u, A, B;
265  double sigma_initial, sigma, sigma_prev, sin_sigma, cos_sigma, cos_2_sigmam, sqr_cos_2_sigmam, delta_sigma;
266  int remaining_steps;
267  double tmp1, phi2, lambda, C, L;
268 
269  /* Check input */
270  NMEA_ASSERT(start_pos != 0);
271  NMEA_ASSERT(end_pos != 0);
272 
273  if (fabs(distance) < 1e-12)
274  { /* No move */
275  *end_pos = *start_pos;
276  if ( end_azimuth != 0 ) *end_azimuth = azimuth;
277  return ! (NMEA_POSIX(isnan)(end_pos->lat) || NMEA_POSIX(isnan)(end_pos->lon));
278  } /* No move */
279 
280  /* Earth geometry */
281  f = NMEA_EARTH_FLATTENING;
282  a = NMEA_EARTH_SEMIMAJORAXIS_M;
283  b = (1 - f) * a;
284  sqr_a = a * a;
285  sqr_b = b * b;
286 
287  /* Calculation */
288  phi1 = start_pos->lat;
289  tan_U1 = (1 - f) * tan(phi1);
290  cos_U1 = 1 / sqrt(1 + tan_U1 * tan_U1);
291  sin_U1 = tan_U1 * cos_U1;
292  s = distance;
293  alpha1 = azimuth;
294  sin_alpha1 = sin(alpha1);
295  cos_alpha1 = cos(alpha1);
296  tan_sigma1 = tan_U1 / cos_alpha1;
297  sigma1 = atan2(tan_U1, cos_alpha1);
298  sin_alpha = cos_U1 * sin_alpha1;
299  sqr_cos_alpha = 1 - sin_alpha * sin_alpha;
300  cos_alpha = sqrt(sqr_cos_alpha);
301  sqr_u = sqr_cos_alpha * (sqr_a - sqr_b) / sqr_b;
302  A = 1 + sqr_u / 16384 * (4096 + sqr_u * (-768 + sqr_u * (320 - 175 * sqr_u)));
303  B = sqr_u / 1024 * (256 + sqr_u * (-128 + sqr_u * (74 - 47 * sqr_u)));
304 
305  /* Initialize iteration */
306  sigma_initial = s / (b * A);
307  sigma = sigma_initial;
308  sin_sigma = sin(sigma);
309  cos_sigma = cos(sigma);
310  cos_2_sigmam = cos(2 * sigma1 + sigma);
311  sqr_cos_2_sigmam = cos_2_sigmam * cos_2_sigmam;
312  delta_sigma = 0;
313  sigma_prev = 2 * NMEA_PI;
314  remaining_steps = 20;
315 
316  while ((fabs(sigma - sigma_prev) > 1e-12) && (remaining_steps > 0))
317  { /* Iterate */
318  cos_2_sigmam = cos(2 * sigma1 + sigma);
319  sqr_cos_2_sigmam = cos_2_sigmam * cos_2_sigmam;
320  sin_sigma = sin(sigma);
321  cos_sigma = cos(sigma);
322  delta_sigma = B * sin_sigma * (
323  cos_2_sigmam + B / 4 * (
324  cos_sigma * (-1 + 2 * sqr_cos_2_sigmam) -
325  B / 6 * cos_2_sigmam * (-3 + 4 * sin_sigma * sin_sigma) * (-3 + 4 * sqr_cos_2_sigmam)
326  ));
327  sigma_prev = sigma;
328  sigma = sigma_initial + delta_sigma;
329  remaining_steps --;
330  } /* Iterate */
331 
332  /* Calculate result */
333  tmp1 = (sin_U1 * sin_sigma - cos_U1 * cos_sigma * cos_alpha1);
334  phi2 = atan2(
335  sin_U1 * cos_sigma + cos_U1 * sin_sigma * cos_alpha1,
336  (1 - f) * sqrt(sin_alpha * sin_alpha + tmp1 * tmp1)
337  );
338  lambda = atan2(
339  sin_sigma * sin_alpha1,
340  cos_U1 * cos_sigma - sin_U1 * sin_sigma * cos_alpha1
341  );
342  C = f / 16 * sqr_cos_alpha * (4 + f * (4 - 3 * sqr_cos_alpha));
343  L = lambda -
344  (1 - C) * f * sin_alpha * (
345  sigma + C * sin_sigma *
346  (cos_2_sigmam + C * cos_sigma * (-1 + 2 * sqr_cos_2_sigmam))
347  );
348 
349  /* Result */
350  end_pos->lon = start_pos->lon + L;
351  end_pos->lat = phi2;
352  if ( end_azimuth != 0 )
353  {
354  *end_azimuth = atan2(
355  sin_alpha, -sin_U1 * sin_sigma + cos_U1 * cos_sigma * cos_alpha1
356  );
357  }
358  return ! (NMEA_POSIX(isnan)(end_pos->lat) || NMEA_POSIX(isnan)(end_pos->lon));
359 }
